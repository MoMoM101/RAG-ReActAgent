```python
class WebSearchTool(BaseTool):
    name = "web_search"
    description = (
        "Search the internet for information. "
        "Use when the knowledge base does not have enough information to answer the user's question."
    )
    parameters = {
        "type": "object",
        "properties": {
            "query": {"type": "string", "description": "Search query"},
            "max_results": {"type": "integer", "default": 5},
        },
        "required": ["query"],
    }
    max_retries = 2
    retry_strategy = "exponential"

    async def execute(self, query: str, max_results: int = 5) -> ToolResult:
        from config import settings

        if not settings.web_search_enabled:
            return ToolResult(success=False, error="联网搜索未启用")

        try:
            from duckduckgo_search import DDGS

            proxies = settings.web_search_proxy or None
            ddgs = DDGS(proxies=proxies) if proxies else DDGS()

            results = []
            with ddgs:
                for r in ddgs.text(query, max_results=min(max_results, settings.web_search_max_results)):
                    results.append({
                        "title": r.get("title", ""),
                        "snippet": r.get("body", ""),
                        "url": r.get("href", ""),
                    })

            return ToolResult(
                success=True,
                data={"count": len(results), "results": results},
            )
        except ImportError:
            return ToolResult(success=False, error="duckduckgo-search 未安装，请执行 pip install duckduckgo-search")
        except Exception as e:
            return ToolResult(success=False, error=str(e))
```

- [ ] **Step 3: 创建 preprocess.py**

```python
import cv2
import numpy as np


def preprocess(image: np.ndarray, dpi: int = 200) -> np.ndarray:
    """Preprocess image for OCR: grayscale -> denoise -> binarize.
    
    Args:
        image: BGR numpy array
        dpi: target DPI, image is upscaled if below this
    Returns:
        Preprocessed binary image
    """
    # Ensure minimum DPI by upscaling if needed
    h, w = image.shape[:2]
    min_pixels = dpi * 11  # ~11 inches for A4
    if max(h, w) < min_pixels:
        scale = min_pixels / max(h, w)
        image = cv2.resize(image, (int(w * scale), int(h * scale)))

    # Grayscale
    if len(image.shape) == 3:
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    else:
        gray = image

    # Denoise
    denoised = cv2.fastNlMeansDenoising(gray, h=10)

    # Adaptive threshold binarization
    binary = cv2.adaptiveThreshold(
        denoised, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C, cv2.THRESH_BINARY, 11, 2
    )

    return binary


def image_from_bytes(data: bytes, dpi: int = 200) -> np.ndarray:
    """Load image from bytes and preprocess for OCR."""
    arr = np.frombuffer(data, np.uint8)
    img = cv2.imdecode(arr, cv2.IMREAD_COLOR)
    return preprocess(img, dpi=dpi)
```

- [ ] **Step 4: 创建 paddle_ocr.py**

```python
import numpy as np
from .base import BaseOCR
from .preprocess import preprocess


class PaddleOCREngine(BaseOCR):
    def __init__(self):
        from paddleocr import PaddleOCR
        self._ocr = PaddleOCR(lang="ch")

    def recognize(self, image) -> str:
        """Recognize text from preprocessed image."""
        result = self._ocr.ocr(image, cls=False)
        if not result or result[0] is None:
            return ""
        lines = []
        for line in result[0]:
            text = line[1][0]
            if text.strip():
                lines.append(text)
        return "\n".join(lines)

    def recognize_from_bytes(self, data: bytes, dpi: int = 200) -> str:
        """Recognize text from raw image bytes."""
        from .preprocess import image_from_bytes
        processed = image_from_bytes(data, dpi=dpi)
        return self.recognize(processed)
```

```python
import fitz  # PyMuPDF

def _load_pdf(file_path: str) -> str:
    from config import settings

    doc = fitz.open(file_path)
    texts = []
    for page in doc:
        text = page.get_text()
        if len(text.strip()) < settings.ocr_min_text_length and settings.ocr_enabled:
            from ocr.factory import create_ocr
            ocr = create_ocr()
            if ocr:
                pix = page.get_pixmap(dpi=200)
                img_bytes = pix.tobytes("png")
                text = ocr.recognize_from_bytes(img_bytes, dpi=200)
        texts.append(text)
    doc.close()
    return "\n\n".join(texts)
```
